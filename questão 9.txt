#include <stdio.h>
int main(){

dis = (preço*0.12);
preçof = preço + imp + dis;
printf("O preço desse veículo é:%4.2f", preçof);
printf("O imposto sobre o veículo é:%4.2f", imp);
printf("O lucro do distruibuidor é:%4.2f", dis);
return 0;
}

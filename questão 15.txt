#include <stdio.h>
#include <math.h>

int main(void) {
  int a, b, c, delta;
  float x1, x2;
  printf("Oi tudo bom? digite os valores de a, b e c respectivamente:");
  scanf("%d", &a);
  scanf("%d", &b);
  scanf("%d", &c);
  delta=((b*b)-(4*a*c));
  printf("delta = %d, agora vou calcular as raizes caso delta seja maior ou igual a zero\n", delta);
  if(delta>=0){
    x1=(-b - sqrt(delta))/(2*a);
    x2 = (-b + sqrt(delta))/(2*a);
    printf("As raizes são %4.2f e %4.2f", x1, x2);
  }
  else{
    printf("As raizes não existem");
  }
  return 0;
}
